//cadastro usuario
const sequelize = require("sequelize");
const connection = require("./database/connection");

const Usuario = connection.define("usuarios", {
  email: {
    type: sequelize.STRING,
    allowNull: false,
  },
  senha: {
    type: sequelize.STRING,
    allowNull: false,
  },
  senha2: {
    type: sequelize.STRING,
    allowNull: false,
  },
});

Usuario.sync({ force: false }).then(() => {});
module.exports = Usuario;
